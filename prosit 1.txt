#include <iostream>
#include <math.h>
int main()
{
	int Xa, Xb, Ya, Yb ;
	int distance;
	std::cout << "Entrez la valeur du x du premier point : ";
	std::cin >> Xa ;
	std::cout << "Entrez la valeur du y du premier point : ";
	std::cin >> Ya;
	std::cout << "Entrez la valeur du x du deuxieme point : ";
	std::cin >> Xb;
	std::cout << "Entrez la valeur du y du deuxieme point : ";
	std::cin >> Yb;
	distance = sqrt((Xb - Xa) ^ 2 + (Yb - Ya) ^ 2);
	std::cout << "la distance de ces deux points sont de : " << distance;


	return 0;
}
